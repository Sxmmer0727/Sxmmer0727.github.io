MySQL常用单行函数

# 进阶4 ：常用函数介绍
/*
select 函数名(实参列表) 【from 表】

	1.单行函数
	
	2.分组函数(聚合函数)
	
*/
USE myemployees;
#1. 字符函数
#length 获取参数值的字节个数
SELECT LENGTH('john');
#concat 拼接字符串
SELECT CONCAT(last_name,'_',first_name) 姓名 FROM employees;
#upper\lower
SELECT UPPER(last_name) FROM employees;
#substr\substring 索引从1开始(字符长度)
SELECT SUBSTR('123456',1,3) out_put;

#姓名中用首字符大写，其他字符小写，然后用_拼接，显示出来
SELECT 
  CONCAT(
    UPPER(SUBSTR(LAST_NAME, 1, 1)),
    LOWER(SUBSTR(LAST_NAME, 2)),
    '_',
    LOWER(first_name)
  ) 
FROM
  employees ;

#instr 返回字串第一次出现的起始索引，若没有，则返回0
SELECT INSTR('123456','4');
#trim() 去除前后空格
SELECT TRIM('  123456   ') AS out_put;
SELECT TRIM('1' FROM '123456111') AS output;
#lpad 用指定的字符实现左填充指定长度/ rpad（右填充）
SELECT LPAD('123456',10,'*') AS output;
#replace用指定的字符替换指定字符
SELECT REPLACE('123456','1','9') AS output;

#2.数学函数
#round 四舍五入 第二个参数为小数点后保留多少位数
SELECT ROUND(-1.454,2);
#ceil向上取整 返回大于等于该参的最小整数 floor(向下取整)
SELECT CEIL(1.00);
#truncate（截断）,直接舍弃后续小数
SELECT TRUNCATE(1.699,1);
#mod取余数,余数符号与被除数一致
SELECT MOD(-10,3);

#3.日期函数
#now返回系统当前日期和时间
SELECT NOW();
#curdate返回当前日期
SELECT CURDATE();
#curtime返回当前时间
SELECT CURTIME();
#可以获取指定的部分
SELECT YEAR(hiredate),MONTH(NOW()) FROM employees;
#将字符串解析为日期str_to_date date_format
# Y:四位数年份 y:两位数年份 m；两位数月份 c：一位数月份
# h:小时 i分钟 s秒
SELECT STR_TO_DATE('1998-12-13','%Y-%m-%d');
SELECT DATE_FORMAT(NOW(),'%Y年%m月%d日--%H:%i%s');

#4.其他函数
SELECT VERSION();
SELECT DATABASE();
SELECT USER();

#5.流程控制函数
# if-else
SELECT last_name,commission_pct,IF(commission_pct IS NULL ,'0','1') FROM employees;
# case when then else end
#查询员工的工资
SELECT 
  CASE
    department_id 
    WHEN 30 
    THEN salary * 1.1 
    WHEN 40 
    THEN salary * 1.2 
    WHEN 50 
    THEN salary * 1.3 
    ELSE salary 
  END 
  AS 新工资
  FROM employees;
  
SELECT salary,
  CASE
    WHEN salary >20000 THEN 'A'
    WHEN salary >10000 THEN 'B'
    WHEN salary >0 THEN 'C'
    ELSE 'D'
    END
    AS 工资级别
    FROM employees;
